numbers <- c(1, 3, 5, 7, 9)
compute_stats <- function(x) {
  x <- x[x > 0]
  total <- sum(x)
  n <- length(x)
  avg <- mean(x)
  list(total=total, avg=avg, n=n)
}
res <- compute_stats(numbers)
