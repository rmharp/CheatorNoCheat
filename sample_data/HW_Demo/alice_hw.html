# compute summary stats
numbers <- c(1, 2, 3, 4, 5, 6, 7)
mean_val <- mean(numbers)
sd_val <- sd(numbers)
print(paste("mean=", mean_val, "sd=", sd_val))

compute_stats <- function(x) {
  x <- x[x > 0]
  total <- sum(x)
  avg <- mean(x)
  n <- length(x)
  return(list(total=total, avg=avg, n=n))
}

res <- compute_stats(numbers)
print(res)
